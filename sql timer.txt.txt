CREATE EVENT UpdateBookStatus
ON SCHEDULE EVERY 1 HOUR
DO
BEGIN
    UPDATE book_status
    SET status = 'Unreturned'
    WHERE status = 'On-Hand' AND date_return < NOW() - INTERVAL 24 HOUR;
END $$
